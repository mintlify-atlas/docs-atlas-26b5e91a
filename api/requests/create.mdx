---
title: 'Create Request'
api: 'POST /request/register'
description: 'Register a new event request from a client or partner'
---

Creates a new request with contact details, event dates, partner type, and selected rate. Upon successful creation, the system sends confirmation emails to both the client and the administrator.

## Request Body

<ParamField body="v_tittle" type="string" required>
  Event title (4-10 characters)
  
  Example: `"Evento"`
</ParamField>

<ParamField body="v_description" type="string" required>
  Event description (1-500 characters)
  
  Example: `"Descripción del evento o solicitud."`
</ParamField>

<ParamField body="v_name" type="string" required>
  Client name (1-20 characters)
  
  Example: `"Juan Perez"`
</ParamField>

<ParamField body="v_email" type="string" required>
  Client email address
  
  Example: `"cliente@example.com"`
</ParamField>

<ParamField body="v_phone_number" type="string" required>
  Phone number (10-13 characters)
  
  Example: `"3001234567"`
</ParamField>

<ParamField body="v_is_partner" type="boolean" required>
  Whether the requester is a partner
  
  Example: `false`
</ParamField>

<ParamField body="v_pax" type="integer" required>
  Number of attendees (minimum 1)
  
  Example: `50`
</ParamField>

<ParamField body="v_init_date" type="string" required>
  Event start date and time (ISO 8601 format)
  
  Example: `"2025-12-10T10:00:00.000Z"`
</ParamField>

<ParamField body="v_end_date" type="string" required>
  Event end date and time (ISO 8601 format)
  
  Example: `"2025-12-10T14:00:00.000Z"`
</ParamField>

<ParamField body="v_fk_rate" type="integer" required>
  Rate ID (minimum 1)
  
  Example: `2`
</ParamField>

<ParamField body="v_value" type="number" required>
  Estimated event value
  
  Example: `450000`
</ParamField>

## Response

<ResponseField name="message" type="string">
  Success message
  
  Example: `"Request Registrada Exitosamente"`
</ResponseField>

<ResponseField name="info" type="array">
  Array of email addresses that received client confirmation
  
  Example: `["cliente@example.com"]`
</ResponseField>

<ResponseField name="infoAdmin" type="array">
  Array of admin email addresses that received notification
  
  Example: `["admin@example.com"]`
</ResponseField>
