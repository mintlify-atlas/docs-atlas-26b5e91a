---
title: 'List Requests'
api: 'GET /request/get'
description: 'Get all event requests from clients and partners'
---

Retrieves the complete list of requests submitted by clients or partners. Includes detailed information such as personal data, event dates, current status, calculated approximate value, selected space, and associated rate.

<Note>
  This endpoint requires authentication via JWT token stored in cookies.
</Note>

## Authentication

<ParamField header="Cookie" type="string" required>
  JWT authentication token
  
  Example: `"token=eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9..."`
</ParamField>

## Response

<ResponseField name="result" type="array">
  Array of request objects
  
  <Expandable title="Request Object">
    <ResponseField name="id_request" type="integer">
      Unique request identifier
      
      Example: `2`
    </ResponseField>
    
    <ResponseField name="tittle" type="string">
      Event title
      
      Example: `"CUMPLEAÑOS"`
    </ResponseField>
    
    <ResponseField name="description" type="string">
      Event description
      
      Example: `"QUIERO CELEBRAR MI CUMPLE"`
    </ResponseField>
    
    <ResponseField name="name" type="string">
      Client name
      
      Example: `"JULIAN"`
    </ResponseField>
    
    <ResponseField name="email" type="string">
      Client email
      
      Example: `"cliente@gmail.com"`
    </ResponseField>
    
    <ResponseField name="phone_number" type="string">
      Client phone number
      
      Example: `"3108880996"`
    </ResponseField>
    
    <ResponseField name="is_partner" type="boolean">
      Whether the client is a partner
      
      Example: `true`
    </ResponseField>
    
    <ResponseField name="pax" type="integer">
      Number of attendees
      
      Example: `15`
    </ResponseField>
    
    <ResponseField name="init_date" type="string">
      Event start date and time (ISO 8601)
      
      Example: `"2025-11-22T11:30:00.000Z"`
    </ResponseField>
    
    <ResponseField name="end_date" type="string">
      Event end date and time (ISO 8601)
      
      Example: `"2025-11-22T17:30:00.000Z"`
    </ResponseField>
    
    <ResponseField name="status" type="string">
      Request status (PENDIENTE or EN PROGRESO)
      
      Example: `"PENDIENTE"`
    </ResponseField>
    
    <ResponseField name="value_aprox" type="string">
      Approximate value in decimal format
      
      Example: `"250000.00"`
    </ResponseField>
    
    <ResponseField name="espacio" type="string">
      Selected space name
      
      Example: `"Salón Principal"`
    </ResponseField>
    
    <ResponseField name="tarifa" type="string">
      Rate name
      
      Example: `"Tarifa Base"`
    </ResponseField>
  </Expandable>
</ResponseField>
