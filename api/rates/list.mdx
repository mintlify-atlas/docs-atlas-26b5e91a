---
title: 'List Rates'
api: 'GET /rate/get'
description: 'Retrieve all registered rates'
---

This endpoint returns all rates registered in the system. No authentication is required.

<Note>
  This endpoint does not require authentication and can be accessed publicly.
</Note>

## Response

<ResponseField name="result" type="array">
  Array of rate objects
  
  <Expandable title="Rate Object Properties">
    <ResponseField name="id_rate" type="number">
      Unique identifier for the rate
    </ResponseField>
    
    <ResponseField name="name" type="string">
      Name of the rate
    </ResponseField>
    
    <ResponseField name="pax" type="number">
      Maximum capacity (number of people)
    </ResponseField>
    
    <ResponseField name="value4" type="number">
      Price for 4-hour rental period
    </ResponseField>
    
    <ResponseField name="value8" type="number">
      Price for 8-hour rental period
    </ResponseField>
    
    <ResponseField name="value_extra" type="number">
      Price per additional hour
    </ResponseField>
    
    <ResponseField name="isPartner" type="boolean">
      Whether this rate is for partner accounts
    </ResponseField>
    
    <ResponseField name="id_space" type="number">
      ID of the space this rate belongs to
    </ResponseField>
  </Expandable>
</ResponseField>
